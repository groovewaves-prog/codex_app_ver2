# secure_review runtime.

streamlit>=1.33
# PDF text extraction. Without this, the extractor falls back to `pdftotext`

pypdf>=4.0
# R-M (custom mask dictionary) Phase 1: Japanese NER for proper-noun
# candidate extraction. Switched from GiNZA to spacy-official ja_core_news_md
# after the GiNZA boot loop on 2026-05-01: ginza 5.2.0 expects spacy 3.7.x,
# which has no binary wheels for Python 3.14 (Streamlit Cloud's default).
# The spacy-official Japanese model tracks current spacy releases so it
# works on Python 3.14 without needing to delete and re-deploy the app.
# Accuracy on company-name detection is somewhat lower than GiNZA, but the
# trade-off is worth keeping the existing deployment intact.
spacy>=3.8
